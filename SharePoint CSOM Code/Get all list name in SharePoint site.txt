namespace CSOMConsole {     class Program     {         static void Main(string[] args)         {             string userName = "ratestft.com";             Console.WriteLine("Enter your password.");             SecureString password = GetPassword();             // ClienContext - Get the context for the SharePoint Online Site               // SharePoint site URL - https://ravihrepoint.com              using (var clientContext = newClientContext("https://rai.sharepoint.com/sites/MyCSOM"))             {                 // SharePoint Online Credentials                   clientContext.Credentials = newSharePointOnlineCredentials(userName, password);                 // Get the SharePoint web                   Web web = clientContext.Web;                 // Get All list names                 ListCollection listcoll = web.Lists;                 clientContext.Load(listcoll);                 clientContext.ExecuteQuery();                 foreach (varlistitems in listcoll)                 {                     Console.WriteLine("List Title:" + listitems.Title);                     List list = web.Lists.GetByTitle("MyTest");                     ListItemCreationInformation Licm = newListItemCreationInformation();                     ListItem listitem = list.AddItem(Licm);                     listitem["Title"] = listitems.Title;                     listitem.Update();                     clientContext.ExecuteQuery();                 }             }         }         private static SecureString GetPassword()         {             ConsoleKeyInfo info;             //Get the user's password as a SecureString               SecureString securePassword = new SecureString();             do             {                 info = Console.ReadKey(true);                 if (info.Key != ConsoleKey.Enter)                 {                     securePassword.AppendChar(info.KeyChar);                 }             }             while (info.Key != ConsoleKey.Enter);             returnsecurePassword;         }     } }
